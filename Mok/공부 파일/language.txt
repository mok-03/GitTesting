async static private void MyMethodAsync(int count)
    {
        Debug.Log("C");
        Debug.Log("D");
        //여기서 다른 스레드? 로 넘어감
        await Task.Run(async () => //함수는 이전 함수가 끝난뒤 넘어가는거 같음
        {
            for (int i = 1; i <= count; i++)
            {
                Debug.Log($"{i}/{count} ...");
                await Task.Delay(10000); // Thread.Sleep()의 비동기 버전
            }
        });
        // await 함수 끝나면 실행
        Debug.Log("G");

    static void Caller()
    {
        Debug.Log("A");
        Debug.Log("B");

        MyMethodAsync(12); //함수실행

        Debug.Log("E");
        Debug.Log("F");
    }

    // Start is called before the first frame update
    void Start()
    {
        Caller();
    }

    int c; //필드 변수 (전역변수,멤버변수) class 내부에서 사용가능 또는 public 이면 다른class 에서도 class가 있을경우 사용가능
    void Start()
    {
        int a = 10; //지역변수 지역에서만 사용가능

        valFunc(a); //call by value  20
        Debug.Log(a);              //   10

        c = new int();
        c = 10;
        RefFunc(ref c);//cal by refurence //20
        fildfunc();                       //21
        Debug.Log(c);

        List<int> K = new List<int>(); 
        K.Add(10);Debug.Log(K[0]);              //10
        ListFunc( K); //List by Refurence         20
        Debug.Log(K[0]);                        //20 list는 기본적으로 ref

    }

    void valFunc(int _a)
    {
        _a = 20;
        Debug.Log(_a);

    }
    void RefFunc(ref int _a)
    {
        _a = 20;
        Debug.Log(_a);
    }
    void fildfunc()
    {
        c++; //필드(전역) 변수라서 클래스 내부의 함수에서 수정가능 
    }
    void ListFunc( List<int> _a)
    {
        _a[0] = 20;
        Debug.Log(_a[0]);

    }
}
